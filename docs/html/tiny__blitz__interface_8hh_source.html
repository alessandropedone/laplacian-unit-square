33template <class real, int SIZE>
34class tiny_blitz_interface {
35 public:

38 typedef std::vector<real> stl_vector;
39 typedef std::vector<stl_vector> stl_matrix;

41 typedef TinyVector<real, SIZE> gene_vector;
42 typedef TinyMatrix<real, SIZE, SIZE> gene_matrix;
43
44 static inline std::string name() { return "tiny_blitz"; }
45
46 static void free_matrix(gene_matrix& A, int N) {}
47
48 static void free_vector(gene_vector& B) {}
49
50 static inline void matrix_from_stl(gene_matrix& A, stl_matrix& A_stl) {
51 for (int j = 0; j < A_stl.size(); j++)
52 for (int i = 0; i < A_stl[j].size(); i++) A(i, j) = A_stl[j][i];
53 }
54
55 static inline void vector_from_stl(gene_vector& B, stl_vector& B_stl) {
56 for (int i = 0; i < B_stl.size(); i++) B(i) = B_stl[i];
57 }
58
59 static inline void vector_to_stl(gene_vector& B, stl_vector& B_stl) {
60 for (int i = 0; i < B_stl.size(); i++) B_stl[i] = B(i);
61 }
62
63 static inline void matrix_to_stl(gene_matrix& A, stl_matrix& A_stl) {
64 int N = A_stl.size();
65 for (int j = 0; j < N; j++) {
66 A_stl[j].resize(N);
67 for (int i = 0; i < N; i++) A_stl[j][i] = A(i, j);
68 }
69 }
70
71 static inline void copy_matrix(const gene_matrix& source, gene_matrix& cible, int N) {
72 for (int j = 0; j < N; j++)
73 for (int i = 0; i < N; i++) cible(i, j) = source(i, j);
74 }
75
76 static inline void copy_vector(const gene_vector& source, gene_vector& cible, int N) {
77 for (int i = 0; i < N; i++) {
78 cible(i) = source(i);
79 }
80 }
81
82 static inline void matrix_matrix_product(const gene_matrix& A, const gene_matrix& B, gene_matrix& X, int N) {
83 X = product(A, B);
84 }
85
86 static inline void matrix_vector_product(gene_matrix& A, gene_vector& B, gene_vector& X, int N) { X = product(A, B); }
87
88 static inline void axpy(const real coef, const gene_vector& X, gene_vector& Y, int N) { Y += coef * X; }
89};
